1.Se requiere obtener el nombre, tipo de documento y la fecha de registro de un solicitante.
2.Se requiere obtener tipo de tramite y medio de respuesta de un solicitante.
3.Se requiere saber la dependencia del tramite 
4.Se requiere saber el nombre y el cargo del responsable del tramite
5. Se requiere saber el estado del tramite y la fecha de finalizacion 

1.select solicitante.nombre, solicitante.Apellidos, solicitante.id_Identificacion from solicitante
where id_solicitante=1
group by solicitante.nombre, solicitante.Apellidos, solicitante.id_Identificacion; 

2. select tramisoft.solicitante.Nombre,tramisoft1.Tramite.catalogo_tramite_idTramite,tramisoft1.Tramite.medio_respuesta
from tramisoft.solicitante, tramisoft1.tramite
where tramisoft.solicitante.id_Solicitante=tramisoft1.Tramite.solicitante_id_Solicitante
group by tramisoft.solicitante.Nombre , tramisoft1.Tramites.catalogo_tramite_idTramite,tramisoft1.Tramite.medio_respuesta

3. Se requiere saber la dependencia del tramite 
   select tramisoft.solicitante.Nombre,dependencia.catalogo_dependencia

from tramisoft.solicitante, tramisoft.dependencia,tramisoft1.tramite

where tramisoft.solicitante.id_Solicitante=tramisoft1.Tramite.solicitante_id_Solicitante and tramisoft1.Tramite.idTramite=tramisoft.dependencia.Tramite_idTramite
group by tramisoft.solicitante.Nombre,dependencia.catalogo_dependencia;


4. Se requiere saber el nombre y el cargo del responsable del tramite

select responsable_tramite.Responsable_cargo,responsable_tramite.Responsable_cargo
from tramisoft.responsable_tramite,Responsable,Cargo
where Responsable.idResponsable=responsable_tramite.Responsable_cargo and Cargo.Cargocol=responsable_tramite=Cargo_responsable and 
Responsable.nombre=responsable_tramite.Responsable_cargo and Cargo.idCargo=responsable_tramite.Cargo_responsable
group by responsable_tramite.Responsable_cargo,responsable_tramite.Cargo_Responsable;

5. Se requiere saber el nombre, tipo de tramite  el resposable,  estado del tramite y la fecha de finalizacion.

select tramisoft.solicitante.Nombre,tramisoft1.Tramite.catalogo_tramite_idTramite,responsable_tramite.Responsable_cargo,tramisoft1.seguimiento.estado_etapa_id_Estado_Etapa,
tramisoft1.Seguimiento.fecha
from tramisoft.solicitante, tramisoft1.tramite,tramisoft.responsable_tramite,tramisoft1.seguimiento, tramisoft1.revision
where tramisoft.solicitante.id_Solicitante=tramisoft1.Tramite.solicitante_id_Solicitante and responsable_tramite.id_Responsable_T=tramisoft1.revision.responsable_tramite_id_Responsable_T
and tramisoft1.revision.idRevision=seguimiento.Revision_idRevision
group by tramisoft.solicitante.Nombre,tramisoft1.Tramite.catalogo_tramite_idTramite,responsable_tramite.Responsable_cargo,tramisoft1.seguimiento.estado_etapa_id_Estado_Etapa,
Seguimiento.fecha;
